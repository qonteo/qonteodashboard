import { fetchConToken } from "../helpers/fetch";
import { types } from "../types/types"

/**
 * @module Actions
 */

/**
 * funcion para traer los datos de las tablas
 * @returns callback con los datos inicializados de la tabla
 */
const getTable = () => {
    
    return async (dispatch,getState) => {
        const { groups } = getState().auth;
        const groupId = groups.filter(g => g.code === 'EC.MDLZ.01')[0].id;
        const resp = await (await fetchConToken(`/mondelez/get-resource-tracking`)).json();
        const { resources, hours } = resp

        dispatch(initTable({
            resources,
            hours
        }))
    }
}

/**
 * funcion para inicializar la tabla
 * @param {Object} data los resources y los hours
 */
const initTable = (data) => ({
    type: types.initializeTable,
    payload: data
})

export {
    getTable
}
