/**
 * Reducer de datos para las tablas por categorias
 * @param {Object} state Objecto de estado de los datos las tablas
 * @param {Object} action action Objecto para cambiar el estado de los las tablas
 * @returns {Object} Estado actual de los las tablas
 */

const tableReducer = (state = initialize, action) => {
    switch (action.type) {
        case types.initializeTable:
          return {
            ...state,
            ...action.payload
          }
        default:
          return state;
      }
}

export default tableReducer
